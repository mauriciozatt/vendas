CREATE TABLE CLIENTES(
ID INTEGER PRIMARY KEY NOT NULL, 
NOME VARCHAR(100)
);

CREATE TABLE PRODUTOS(
ID INTEGER PRIMARY KEY NOT NULL,
DESCRICAO VARCHAR(100), 
PRECO NUMERIC(20,2)
);


CREATE TABLE PEDIDOS(
ID INTEGER PRIMARY KEY NOT NULL,
CLIENTE_ID INTEGER NOT NULL,
DATA_PEDIDO TIMESTAMP,
STATUS VARCHAR(20), 
TOTAL NUMERIC(20,2), 
FOREIGN KEY (CLIENTE_ID) REFERENCES CLIENTES (ID)
);  

CREATE TABLE PEDIDOS_ITENS(
ID INTEGER PRIMARY KEY NOT NULL,
PEDIDO_ID INTEGER NOT NULL,
PRODUTO_ID INTEGER NOT NULL,
QUANTIDADE NUMERIC(20,2)
);

ALTER TABLE PEDIDOS_ITENS ADD CONSTRAINT fk1 FOREIGN KEY (PEDIDO_ID) REFERENCES PEDIDOS(ID);

ALTER TABLE PEDIDOS_ITENS ADD CONSTRAINT fk2 FOREIGN KEY (PRODUTO_ID) REFERENCES PRODUTOS(ID);
